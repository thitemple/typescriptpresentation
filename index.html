type alias

type functionOrObject = ()=>{}|{};

function execute(config: functionOrObject) {
	let params: {};
	if(typeof config === "function") {
		params = config();
	} else {
		params = {};
	}
}

function processString(val: string|string[]) {
	if(typeof val === "string") {
		console.log(val);
	} else {
		for(let v of val) {
			console.log(v);
		}
	}	
}
